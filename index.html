function animarCabo(cabo) { fecharBalao(); let frame = 0; const camada = document.getElementById(`camada-${cabo.id.toLowerCase()}`); camada.style.opacity = "1"; // Mudamos o tempo para 15ms para o cabo ir mais rápido e suave const loop = setInterval(() => { if (frame > cabo.max) { clearInterval(loop); return; // Interrompe a execução imediatamente } const frameString = frame.toString().padStart(4, '0'); camada.style.backgroundImage = `url('${cabo.pasta}/quadro${frameString}.png')`; if (frame === cabo.max) { clearInterval(loop); // Limpa o loop aqui somConexao.currentTime = 0; somConexao.play(); if (cabo.id === 'FASE1') { balao.style.display = 'flex'; balaoCor.style.display = 'flex'; } if (cabo.id === 'NEUTRO1') { balaoNeutro.style.display = 'flex'; } if (cabo.id === 'FASE2') { balaoSeccionar.style.display = 'flex'; } if (cabo.id === 'RETORNO1') { balaoRetorno.style.display = 'flex'; } let idWago; if (cabo.id === 'FASE1') idWago = 'wago-fase'; else if (cabo.id === 'NEUTRO1') idWago = 'wago-neutro'; else if (cabo.id === 'RETORNO1') idWago = 'wago-retorno'; else if (cabo.id === 'RETORNO2') idWago = 'wago-retorno'; if (idWago) { setTimeout(() => { const wago = document.getElementById(idWago); wago.style.transform = `translate(-50%, -50%) rotate(${cabo.wagoRot})`; wago.style.display = 'block'; wago.title = "Para interligações consulte em 'AJUDA'"; }, 1500); } etapaAtual++; setTimeout(carregarEtapa, 600); return; // CRÍTICO: Garante que a função morra aqui e não execute o frame++ abaixo } frame += 1; }, 15); // Velocidade otimizada }
